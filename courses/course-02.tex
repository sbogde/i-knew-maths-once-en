\subsection*{Equivalence Relations. Quotient Sets (11-Oct-1994)}

Sets $\rightarrow$ maps (morphisms) of sets $\rightarrow$ subsets $\rightarrow$ quotient sets.

\begin{definition}
A binary relation is a pair $(A, R)$, with $A \neq \emptyset$ and $R \subseteq A \times A$.
\end{definition}

\textbf{Examples:}
\begin{enumerate}
    \item[(1)] $A = \mathbb{N}, \quad R = \{(x, y) \in \mathbb{N} \times \mathbb{N} \mid (\exists) u \in \mathbb{N} \text{ such that } x + u = y\}$
    \item[(2)] $A = \mathbb{N}, \quad R = \{(x, y) \in \mathbb{N} \times \mathbb{N} \mid (\exists) q \in \mathbb{N} \text{ such that } x \cdot q = y\}$
    \item[(1)'] $A = \mathbb{Z}, \quad R = \{(x, y) \in \mathbb{Z} \times \mathbb{Z} \mid (\exists) u \in \mathbb{Z} \text{ such that } x + u = y\}$
    \item[(2)'] $A = \mathbb{Z}, \quad R = \{(x, y) \in \mathbb{Z} \times \mathbb{Z} \mid (\exists) q \in \mathbb{N} \text{ such that } x \cdot q = y\}$
    \item[(3)] $A = \mathbb{R}, \quad R = \{(x, y) \in \mathbb{R} \times \mathbb{R} \mid x - y \in \mathbb{Z}\}$
    \item[(4)] $n \in \mathbb{N}^*, \quad A = \mathbb{Z}, \quad R_n = \{(x, y) \in \mathbb{Z} \mid n \mid x - y \}$
    \item[(5)] $A = \text{the set of lines in a Euclidean plane}$, \\ $R = \{(d_1, d_2) \in A \times A \mid d_1 = d_2 \text{ or } d_1 \cap d_2 = \emptyset\}$
    \item[(6)] $A \xrightarrow{f} B, \quad R_f = \{(x, y) \in A \times A \mid f(x) = f(y)\}$
\end{enumerate}

\textbf{Example (with notations from (4) and (6)):} $n \in \mathbb{N}^*$, $f: \mathbb{Z} \rightarrow \mathbb{C}$, $f(k) = \cos \frac{2k\pi}{n} + i \sin \frac{2k\pi}{n}$. Then $R_f = R_n$. \textit{(The proof for this statement is presented at the end of the course).}

If $x, y \in A$ and $(x, y) \in R$, we denote this by $x R y$. If $(x, y) \notin R$, we denote this by $x \not R y$.

\begin{definition}
Let $R$ be a binary relation on $A$. We say the binary relation is:
\begin{itemize}
    \item \textbf{reflexive} if $(\forall) x \in A \implies x R x$.
    \item \textbf{symmetric} if $x R y \implies y R x$.
    \item \textbf{antisymmetric} if $x R y$ and $y R x \implies x = y$.
    \item \textbf{transitive} if $x R y$ and $y R z \implies x R z$.
\end{itemize}
\end{definition}

\begin{definition}
A binary relation $R$ on $A$ which is:
\begin{enumerate}
    \item reflexive, symmetric, and transitive is called an \textbf{equivalence relation}.
    \item reflexive, antisymmetric, and transitive is called an \textbf{order relation}.
    \item reflexive and transitive is called a \textbf{preorder (quasi-order)}.
\end{enumerate}
\end{definition}

\textbf{Examples:}
\begin{itemize}
    \item \textbf{Equivalence relation:} Congruence modulo $n$ on $\mathbb{Z}$, i.e., $(\mathbb{Z}, \equiv \pmod n)$.
    \item \textbf{Order relation:} Set inclusion $\subseteq$ on the power set of a set, $(\mathcal{P}(X), \subseteq)$.
    \item \textbf{Preorder relation:} Divisibility on $\mathbb{Z}$. It is reflexive and transitive, but fails to be antisymmetric since $a \mid b$ and $b \mid a \implies a = \pm b$.
\end{itemize}

Concepts associated with an equivalence relation: \textbf{equivalence class} and \textbf{complete system of representatives (transversal)}.

\begin{definition}
Let $A \neq \emptyset$ be a set equipped with an equivalence relation. A subset $\emptyset \neq C \subseteq A$ is called an \textbf{equivalence class} if:
\begin{enumerate}
    \item $(\forall) x, y \in C \implies x \sim y$
    \item if $z \sim x$, $x \in A$, and $x \in C \implies z \in C$
\end{enumerate}
\end{definition}

\textbf{Example:} $n = 6$, $C = 6\mathbb{Z} + 1 \subset \mathbb{Z}$. \\
$R_6 = \{(x, y) \in \mathbb{Z} \times \mathbb{Z} \mid 6 \mid x - y\}$
\begin{enumerate}
    \item $x, y \in C \implies 6 \mid x - y$
    \item $z \equiv x \pmod 6 \implies z - x = 6q_1$. Knowing that $x = 6q + 1$, we get: \\
    $z - (6q + 1) = 6q_1 \implies z = 6(q + q_1) + 1 \in C$.
\end{enumerate}

Let $(A, R)$ be given and $a \in A$. We denote $[a]_R = \hat{a} = C_a = \{x \in A \mid x \sim a\} \subseteq A$.

\begin{proposition}
\begin{enumerate}
    \item $\hat{a}$ is an equivalence class.
    \item If $C \subseteq A$ is an equivalence class, then $C = [a]_R \quad (\forall) a \in C$.
\end{enumerate}
\end{proposition}

\begin{proof}
1) $\emptyset \neq C_a$ because $a \in C_a$ (by reflexivity $a \sim a$). \\
$(\forall) x, y \in C_a \implies x \sim a$ and $y \sim a \implies x \sim y$ (by symmetry and transitivity). \\
Assume $z \sim x$, $z \in A$, and $x \in C_a \implies x \sim a \implies z \sim a \implies z \in C_a$.

2) ``$\subseteq$'': Let $x \in C$. Since $a \in C$, we have $x \sim a \implies x \in C_a \implies C \subseteq C_a$ (the equivalence class of $a$). \\
``$\supseteq$'': Let $x \in C_a \implies x \sim a$. Since $a \in C$, we obtain $x \in C \implies C_a \subseteq C$.
\end{proof}

\begin{proposition}
Let $(A, R)$ be a set with an equivalence relation $R$.
\begin{enumerate}
    \item For $a, b \in A$, we have $\hat{a} = \hat{b} \iff a \sim b$.
    \item If $C'$ and $C''$ are two equivalence classes, then either $C' = C''$ or $C' \cap C'' = \emptyset$.
\end{enumerate}
\end{proposition}

\vspace{1em}
\textbf{Proof continuation:}
\begin{proof}
(1)
``$\implies$'': $\hat{a} = \hat{b} \implies a \in \hat{a} \implies a \in \hat{b} \implies a \sim b$. \\
``$\impliedby$'': $a \sim b \overset{?}{\implies} \hat{a} = \hat{b}$. \\
``$\subseteq$'': Let $x \in \hat{a} \implies x \sim a$. Since $a \sim b \implies x \sim b \implies x \in \hat{b}$. \\
``$\supseteq$'': Analogous (idem).

(2) Let $c \in C' \cap C'' \implies C' = \hat{c} = C''$.
\end{proof}

\begin{definition}
Let $(A, R)$ with $A \neq \emptyset$ and $R$ an equivalence relation. A family of elements from $A$ is called a \textbf{complete system of representatives} (transversal) if:
\begin{itemize}
    \item[$(\alpha)$] $(\forall) i, j \in I, i \neq j \implies a_i \not\sim a_j$
    \item[$(\beta)$] $(\forall) x \in A, (\exists!) i \in I \text{ s.t. } x \sim a_i$
\end{itemize}
\end{definition}

\textbf{Examples:}
\begin{enumerate}
    \item $(\mathbb{Z}, R_6)$, the set $\{0, 1, 2, 3, 4, 5\} \subset \mathbb{Z}$ is a complete system of representatives for $R_6$. \\
    $x \in \mathbb{Z} \implies x = 6q + r, \quad 0 \leq r < 6$. \\
    The set $\{6q, 1+6q, \dots, 5+6q\}$ is also a transversal. \\
    $x - r \text{ is a multiple of } 6 \implies x \equiv r \pmod 6$.
    \item $A = \mathbb{R}$. The interval $[0, 1) \subset \mathbb{R}$, and $R = \{(x, y) \in \mathbb{R} \times \mathbb{R} \mid x - y \in \mathbb{Z}\}$. \\
    $x \in \mathbb{R} \implies (\exists!) a \in [0, 1) \text{ s.t. } x - a \in \mathbb{Z}$. \\
    $x = [x] + \{x\}$, where $a = \{x\} \in [0, 1) \implies [0, 1)$ is a complete system of representatives.
\end{enumerate}

\begin{definition}
Let $(A, R)$, where $R$ is an equivalence relation. The set:
$$ A/R = \{\hat{a} \mid a \in A\} $$
represents the set of all equivalence classes and is called the \textbf{quotient set} of $A$ by $R$.
\end{definition}

\begin{proposition}
$A/R = \{\hat{a}_i \mid i \in I\}$ where $\{a_i\}_{i \in I}$ is a complete system of representatives. \\
$(\forall) a \in A, (\exists) a_i$ s.t. $\hat{a} = \hat{a}_i$.
\end{proposition}

\textbf{Examples:}
$\mathbb{Z}/R_6 = \{\hat{0}, \hat{1}, \hat{2}, \hat{3}, \hat{4}, \hat{5}\}$. Generally, $\mathbb{Z}/R_n = \{\hat{0}, \hat{1}, \dots, \widehat{n-1}\}$.

\begin{remark}
If $\{a_i\}_{i \in I} \subset A$ is a transversal in $A$ for an equivalence relation $R$, then:
$$ A = \bigcup_{i \in I} \hat{a}_i \quad \text{(disjoint union)} $$
\end{remark}

\begin{proof}
``$\supseteq$'': Obvious. \\
``$\subseteq$'': We use point $(\beta)$ from the definition of a complete system of representatives. \\
Let $x \in A \implies (\exists) i \in I \text{ s.t. } x \sim a_i \implies x \in \hat{a}_i \implies x \in \bigcup_{i \in I} \hat{a}_i$. \\
The union is disjoint because $\hat{a}_i \cap \hat{a}_j = \emptyset$ for $i \neq j$.
\end{proof}

\textbf{Example:} $\mathbb{Z} = 6\mathbb{Z} \cup (6\mathbb{Z}+1) \cup \dots \cup (6\mathbb{Z}+5)$.

\begin{definition}
A \textbf{partition} of a set $A \neq \emptyset$ is a family of subsets $A_i \subset A$, $A_i \neq \emptyset$, such that:
\begin{itemize}
    \item $i \neq j \implies A_i \cap A_j = \emptyset$
    \item $\bigcup_{i \in I} A_i = A$
\end{itemize}

On a set $A$, we can define as many equivalence relations as there are possible partitions for $A$.
Given a partition, we define the relation: $x, y \in A \implies \left( x \sim y \overset{\text{def}}{\iff} (\exists) i \text{ s.t. } x, y \in A_i \right)$.

\begin{definition}
Let $(A, R)$, where $R$ is an equivalence relation. We define the map:
$$ p : A \rightarrow A/R, \quad p(a) = \hat{a} $$
The map $p$ is surjective and is called the \textbf{canonical projection} to the quotient set. The preimage of an element from the quotient set is an equivalence class.
\end{definition}

\textbf{Example:} $p: \mathbb{Z} \rightarrow \mathbb{Z}/R_n = \mathbb{Z}_n, \quad p(x) = \hat{x} = n\mathbb{Z} + x$.

\begin{theorem}[Factorization theorem for a set map]
(The map induced on a quotient set by a set map). \\
If $f: A \rightarrow B$ is a function compatible with the equivalence relation $R$, then $(\exists) f^* : A/R \rightarrow B$ such that the following triangle commutes ($f = f^* \circ p$).

\begin{center}
\begin{tikzcd}[row sep=large, column sep=large]
A \arrow[r, "f"] \arrow[d, "p"'] & B \\
A/R \arrow[ur, dashed, "f^*"'] &
\end{tikzcd}
\hspace{2cm}
\begin{tikzcd}[row sep=large, column sep=large]
a \arrow[r, mapsto] \arrow[d, mapsto] & f(a) \\
\hat{a} \arrow[ur, mapsto] &
\end{tikzcd}
\end{center}
\end{theorem}

\begin{remark}
If the function $f^*$ exists, it is unique. If $f_1^* \circ p = f_2^* \circ p$, since $p$ is surjective (an epimorphism), it follows that $f_1^* = f_2^*$.
\end{remark}

\begin{theorem}
Let $f: A \rightarrow B$ be a set map, $R$ an equivalence relation on $A$, and $p: A \rightarrow A/R$ the canonical projection.
\begin{enumerate}
    \item $(\exists)$ a map $f^*: A/R \rightarrow B$ such that $f^* \circ p = f \iff R \subseteq R_f$. \\
    (where $R_f = \{(x,y) \in A \times A \mid f(x) = f(y)\}$)
    \item $f^*$ is surjective $\iff f$ is surjective.
    \item $f^*$ is injective $\iff R = R_f$.
    \item $f^*$ is bijective $\iff f$ is surjective and $R = R_f$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $R \subseteq R_f \iff f$ is constant on equivalence classes. \\
``$\implies$'': $(x,y) \in R \implies x \sim y \implies x, y \in C \implies f(x) = f(y) \implies (x,y) \in R_f$. \\
``$\impliedby$'': Assume $(\exists) f^*: A/R \rightarrow B$ such that $f = f^* \circ p$. \\
Let $(x,y) \in R \implies \hat{x} = \hat{y}$. \\
$f(x) = (f^* \circ p)(x) = f^*(p(x)) = f^*(\hat{x}) = f^*(\hat{y}) = f^*(p(y)) = (f^* \circ p)(y) = f(y) \implies (x,y) \in R_f$.

Now, assume $R \subseteq R_f$ and we have the commutative triangle $f^* \circ p = f$. We must assign the rule of correspondence for $f^*$. \\
Let $C \in A/R$. We seek $f^*(C) \in B$. \\
We set $f^*(C) \overset{\text{def}}{=} f(a)$, where $a \in C$. \\
\textit{Well-definedness:} Let $b \in C \implies f^*(C) = f(b)$. \\
$\implies f(a) = f(b)$ is true (since $R \subseteq R_f$). \\
(from $a,b \in C \implies a \sim b \implies (a,b) \in R \implies f(a) = f(b)$). \\
$\implies f^*$ is well-defined. \\
We have $f^*(\hat{a}) = f(a)$ (since $\hat{a} = C$). \\
$(\forall) a \in A \implies f^*(p(a)) = f(a) \implies (f^* \circ p)(a) = f(a) \implies f^* \circ p = f$.

(2) 
``$\implies$'': $f^* \circ p$ surjective $\implies f$ surjective. \\
``$\impliedby$'': $f$ surjective $\implies f^* \circ p$ surjective $\implies f^*$ surjective.

(3) $f^*$ injective $\iff \left( f^*(\hat{x}) = f^*(\hat{y}) \implies \hat{x} = \hat{y} \right)$. \\
Since $f^*(\hat{x}) = f(x)$ and $f^*(\hat{y}) = f(y)$, this is equivalent to: \\
$\iff \left( f(x) = f(y) \implies x \sim y \right) \iff \left( (x,y) \in R_f \implies (x,y) \in R \right)$. \\
``$\implies$'': Assume $R = R_f$. \\
$f^*(\hat{x}) = f^*(\hat{y}) \implies f(x) = f(y) \implies (x,y) \in R_f \implies (x,y) \in R \implies \hat{x} = \hat{y} \implies f^*$ is injective. \\
``$\impliedby$'': Let $(x,y) \in R_f \implies f(x) = f(y) \implies f^*(\hat{x}) = f^*(\hat{y}) \implies \hat{x} = \hat{y} \implies x \sim y \implies (x,y) \in R \implies R_f \subseteq R$. But $R \subseteq R_f$ (by hypothesis), so $R = R_f$.

(4) Obvious (by combining points 2 and 3).
\end{proof}

\vspace{1em}
\textbf{Returning to the example with the map $f_n(k) = \cos \frac{2k\pi}{n} + i \sin \frac{2k\pi}{n}$:} \\
We have $R_{f_n} = \{(h,k) \in \mathbb{Z} \times \mathbb{Z} \mid f_n(k) = f_n(h)\}$.
$$ \cos \frac{2k\pi}{n} + i \sin \frac{2k\pi}{n} = \cos \frac{2h\pi}{n} + i \sin \frac{2h\pi}{n} $$
$$ \implies \frac{2k\pi}{n} - \frac{2h\pi}{n} = 2\pi t, \quad t \in \mathbb{Z} $$
$$ \implies k - h = nt \implies n \mid (k - h) \implies (h,k) \in R_n $$
Thus, $R_{f_n} = R_n$. According to the factorization theorem, $(\mathbb{Z}_n, +) \cong U_n$ (the group of $n$-th complex roots of unity).
